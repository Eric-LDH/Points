### 兑换页面 (exchange.vue)
**路径**: `src/pages/exchange/exchange.vue`

**核心功能**: 周期限制检查

**兑换流程**:
1. 用户点击兑换按钮
2. 检查积分是否足够
3. 检查周期限制
4. 确认兑换
5. 扣除积分并创建记录

**周期检查实现**:
```typescript
import { CycleLimiter } from '@/utils/cycleLimiter'

const canExchange = (item: RewardItem) => {
  if (totalPoints.value < item.points) return false
  
  const result = CycleLimiter.canExchange(
    item,
    exchangeRecords.value,
    store.currentChildId
  )
  
  return result.canExchange
}

const getExchangeButtonText = (item) => {
  const result = CycleLimiter.canExchange(...)
  if (!result.canExchange) {
    return result.message // "本周已兑换 2/2 次"
  }
  return `兑换 (剩余${result.remaining}次)`
}
```

**周期计算逻辑**:
```typescript
// src/utils/cycleLimiter.ts

getCycleStartDate(cycleType, date): Date {
  switch(cycleType) {
    case 'weekly':
      // 返回本周一
      const day = date.getDay() || 7
      const diff = date.getDate() - day + 1
      return new Date(date.setDate(diff))
      
    case 'monthly':
      // 返回本月 1 号
      return new Date(date.getFullYear(), date.getMonth(), 1)
      
    case 'yearly':
      // 返回今年 1 月 1 号
      return new Date(date.getFullYear(), 0, 1)
  }
}
```

## 📊 状态管理 (Pinia Store)

### Store 结构
```typescript
// src/stores/index.ts

export const useAppStore = defineStore('app', () => {
  // State
  const rules = ref<Rule[]>([])
  const rewardItems = ref<RewardItem[]>([])
  const pointsRecords = ref<PointsRecord[]>([])
  const exchangeRecords = ref<ExchangeRecord[]>([])
  const children = ref<Child[]>([])
  const currentChildId = ref<string | null>(null)
  const darkMode = ref(false)
  
  // Getters
  const currentChild = computed(() => 
    children.value.find(c => c.id === currentChildId.value)
  )
  
  const todayPoints = computed(() => {
    const today = new Date().toISOString().split('T')[0]
    return pointsRecords.value
      .filter(r => r.date === today && r.childId === currentChildId.value)
      .reduce((sum, r) => sum + r.points, 0)
  })
  
  // Actions
  function addPointsRecord(record) {
    pointsRecords.value.push(record)
    // 更新孩子积分
  }
  
  function addExchangeRecord(record) {
    exchangeRecords.value.push(record)
    // 扣除积分
  }
})
```
