// ✅ 基于真实记录计算
const weekDays = computed(() => {
  const today = new Date()
  const result = []
  
  // 获取最近 7 天的日期
  for (let i = 6; i >= 0; i--) {
    const date = new Date(today)
    date.setDate(date.getDate() - i)
    const dateStr = date.toISOString().split('T')[0]
    
    // 计算这一天的总积分
    const dayRecords = pointsRecords.value.filter(r => r.date === dateStr)
    const dayPoints = dayRecords.reduce((sum, r) => sum + r.points, 0)
    
    const weekDayNames = ['周日', '周一', '周二', '周三', '周四', '周五', '周六']
    result.push({
      name: weekDayNames[date.getDay()],
      points: dayPoints,
      date: dateStr
    })
  }
  
  return result
})
```

// ✅ 关联规则数据，精确统计每个分类
const categoryStats = computed(() => {
  const rulesMap: Record<string, { icon: string; count: number; points: number }> = {}
  
  // 初始化所有分类
  enabledRules.value.forEach(rule => {
    if (!rulesMap[rule.category]) {
      rulesMap[rule.category] = { icon: rule.icon, count: 0, points: 0 }
    }
  })
  
  // 统计每个分类的记录
  pointsRecords.value.forEach(record => {
    const rule = enabledRules.value.find(r => r.id === record.ruleId)
    if (rule) {
      if (!rulesMap[rule.category]) {
        rulesMap[rule.category] = { icon: rule.icon, count: 0, points: 0 }
      }
      rulesMap[rule.category].count += 1
      rulesMap[rule.category].points += record.points
    }
  })
  
  // 计算百分比
  const totalCount = pointsRecords.value.length
  
  return Object.entries(rulesMap)
    .map(([category, data]) => ({
      category,
      icon: data.icon,
      count: data.count,
      totalPoints: data.points,
      percentage: totalCount > 0 ? Math.round((data.count / totalCount) * 100) : 0
    }))
    .filter(item => item.count > 0)
    .sort((a, b) => b.count - a.count)
})
```

const monthExchange = 150 // TODO: 计算实际值

**修改后**：
```typescript
// ✅ 从兑换记录中计算
const monthExchange = computed(() => {
  const today = new Date()
  const currentMonth = today.toISOString().slice(0, 7) // YYYY-MM
  
  // 筛选本月的兑换记录
  const monthRecords = store.exchangeRecords.filter(r => {
    const recordDate = new Date(r.exchangeDate)
    return recordDate.toISOString().slice(0, 7) === currentMonth
  })
  
  // 计算总兑换积分
  return monthRecords.reduce((sum, r) => sum + r.totalPoints, 0)
})
```

### 3. 首页连续打卡天数真实计算

#### 问题描述
**修改前**：
```typescript
const streakDays = ref(5) // TODO: 计算实际连续天数
```

#### 实现方案

**真实计算逻辑**：
```typescript
const streakDays = computed(() => {
  if (pointsRecords.value.length === 0) return 0
  
  // 获取所有有记录的日期（去重）
  const dates = [...new Set(pointsRecords.value.map(r => r.date))]
  
  // 按日期降序排序
  dates.sort((a, b) => new Date(b).getTime() - new Date(a).getTime())
  
  if (dates.length === 0) return 0
  
  // 检查是否包含今天
  const today = new Date().toISOString().split('T')[0]
  let startIndex = 0
  
  if (dates[0] !== today) {
    // 如果最新记录不是今天，从昨天开始算
    const yesterday = new Date(Date.now() - 86400000).toISOString().split('T')[0]
    if (dates[0] !== yesterday) {
      return 0 // 昨天和今天都没记录，中断了
    }
    startIndex = 0
  }
  
  // 计算连续天数
  let streak = 1
  for (let i = startIndex; i < dates.length - 1; i++) {
    const currentDate = new Date(dates[i])
    const previousDate = new Date(dates[i + 1])
    
    // 计算两个日期之间相差的天数
    const diffDays = Math.floor((currentDate.getTime() - previousDate.getTime()) / 86400000)
    
    if (diffDays === 1) {
      streak++
    } else {
      break // 中断了
    }
  }
  
  return streak
})
```

**计算逻辑说明**：
1. **提取日期**：从所有积分记录中提取日期并去重
2. **排序**：按时间倒序排列
3. **检查连续性**：
   - 如果今天有记录，从今天开始往前推
   - 如果今天没记录但昨天有，从昨天开始往前推
   - 如果今天和昨天都没记录，返回 0（中断）
4. **计算连续天数**：逐日检查，相差 1 天则累加，否则中断

**效果展示**：
